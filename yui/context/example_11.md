```yui
標準ライブラリを使う
# 簡易スタックマシン
# 命令は [opcode, operand] の形の配列で表す。operand が不要な命令は 0 を入れる。
#
# 対応命令:
#   ["PUSH", n] -- スタックに n を積む
#   ["POP",  0] -- スタック頂上を捨てる
#   ["DUP",  0] -- スタック頂上を複製する
#   ["ADD",  0] -- 上 2 つを取り出して a+b を積む
#   ["SUB",  0] -- 上 2 つを取り出して a-b を積む
#   ["MUL",  0] -- 上 2 つを取り出して a*b を積む
#
# プログラム実行後、スタックの頂上を答えとして返す。
# (Yui の配列は要素削除ができないので、論理的なスタックの大きさを top で管理し、
#  pop は top を減らすだけにする)

run=入力programに対し{
   stack=[]
   top=0
   n=programの大きさ
   i=0
   n回くり返す{
      instr=program[i]
      op=instr[0]
      arg=instr[1]

      もしopが"PUSH"ならば{
         # 論理 top が配列実長より小さければ上書き、そうでなければ末尾に追加
         sz=stackの大きさ
         もしtopがszより小さいならば{
            stack[top]=arg
         }
         そうでなければ{
            stackにargを追加する
         }
         topを増やす
      }
      もしopが"POP"ならば{
         topを減らす
      }
      もしopが"DUP"ならば{
         t=差(top,1)
         x=stack[t]
         sz=stackの大きさ
         もしtopがszより小さいならば{
            stack[top]=x
         }
         そうでなければ{
            stackにxを追加する
         }
         topを増やす
      }
      もしopが"ADD"ならば{
         topを減らす
         b=stack[top]
         topを減らす
         a=stack[top]
         stack[top]=和(a,b)
         topを増やす
      }
      もしopが"SUB"ならば{
         topを減らす
         b=stack[top]
         topを減らす
         a=stack[top]
         stack[top]=差(a,b)
         topを増やす
      }
      もしopが"MUL"ならば{
         topを減らす
         b=stack[top]
         topを減らす
         a=stack[top]
         stack[top]=積(a,b)
         topを増やす
      }
      iを増やす
   }
   # スタックの頂上を結果として返す
   topを減らす
   stack[top]が答え
}

# テスト: 3 + 4 = 7
>>> run([["PUSH",3],["PUSH",4],["ADD",0]])
7
# テスト: (3 + 4) * 2 = 14
>>> run([["PUSH",3],["PUSH",4],["ADD",0],["PUSH",2],["MUL",0]])
14
# テスト: 10 - (2 + 3) = 5
>>> run([["PUSH",10],["PUSH",2],["PUSH",3],["ADD",0],["SUB",0]])
5
# テスト: DUP で 5 を複製して掛ける → 25
>>> run([["PUSH",5],["DUP",0],["MUL",0]])
25
# テスト: POP で途中の値を捨てる → 7
>>> run([["PUSH",7],["PUSH",99],["POP",0]])
7
# テスト: 単一 PUSH
>>> run([["PUSH",42]])
42
```

```yui
標準ライブラリを使う
# 複利計算
#
# 元本 P を年利 r で n 年運用したときの最終金額は、いわゆる複利の公式で
#   A は P と (1+r) の n 乗の積
# になる。yui には べき乗 が無いので、毎年 balance を (1+r) 倍するループで実装する。
#
# 利率 r は小数で渡す (例: 5% なら 0.05)。
# 結果は小数で返る。

# === 1. n 年後の元利合計 ===
compound=入力P,r,nに対し{
   balance=小数化(P)
   factor=和(小数化(1),r)
   n回くり返す{
      balance=積(balance,factor)
   }
   balanceが答え
}

# === 2. 元本が 2 倍になるのに何年かかるか (年単位の切り上げ) ===
# 利率 r で運用したとき、balance が初めて元本の 2 倍以上になる年を返す
years_to_double=入力rに対し{
   balance=小数化(1)
   factor=和(小数化(1),r)
   n=0
   # 念のため上限 1000 年でガード
   1000回くり返す{
      もしbalanceが2以上ならば{
         くり返しを抜ける
      }
      balance=積(balance,factor)
      nを増やす
   }
   nが答え
}

# === テスト: 1 年目 ===
>>> compound(1000,0.05,1)
1050.000000
# === テスト: 2 年目 (1000 × 1.05 × 1.05) ===
>>> compound(1000,0.05,2)
1102.500000
# === テスト: 3 年目 (1.10 を 3 回かける) ===
>>> compound(1000,0.10,3)
1331.000000
# === テスト: 利率 0 なら元本のまま ===
>>> compound(100,0.0,5)
100.000000
# === テスト: 0 年なら元本のまま ===
>>> compound(1000,0.05,0)
1000.000000

# === テスト: 年利 10 パーセントなら、8 年目に元本が 2 倍を超える ===
>>> years_to_double(0.10)
8
# === テスト: 年利 7 パーセントなら 11 年 ===
>>> years_to_double(0.07)
11
# === テスト: 年利 5 パーセントなら 15 年 (いわゆる 72 の法則) ===
>>> years_to_double(0.05)
15
```

```yui
標準ライブラリを使う
# ROT13: アルファベットを 13 文字ぶん回転させる古典暗号
# 文字列は内部的に文字コード配列なので 1 文字ずつコード値を取り出して変換する
# 文字コード: 'A'=65, 'Z'=90, 'a'=97, 'z'=122
# ROT13 は自己逆 (2 回適用すると元に戻る) なので暗号化と復号化は同じ関数で済む

rot13=入力sに対し{
   result=""
   n=sの大きさ
   i=0
   n回くり返す{
      c=s[i]
      nc=c
      # 大文字 A..Z
      もしcが65以上ならば{
         もしcが90以下ならば{
            offset=差(c,65)
            shifted=剰余(和(offset,13),26)
            nc=和(shifted,65)
         }
      }
      # 小文字 a..z
      もしcが97以上ならば{
         もしcが122以下ならば{
            offset=差(c,97)
            shifted=剰余(和(offset,13),26)
            nc=和(shifted,97)
         }
      }
      resultにncを追加する
      iを増やす
   }
   resultが答え
}

# テスト: 大文字の基本ケース  ABC → NOP
>>> rot13("ABC")
"NOP"
# テスト: 小文字の基本ケース  xyz → klm (折り返し)
>>> rot13("xyz")
"klm"
# テスト: 混在 + 記号 + 数字 (アルファベット以外はそのまま)
>>> rot13("Hello, World!")
"Uryyb, Jbeyq!"
# テスト: 自己逆性  rot13(rot13(x)) == x
>>> rot13("Uryyb, Jbeyq!")
"Hello, World!"
# テスト: 数字・記号は不変
>>> rot13("123 !?")
"123 !?"
# テスト: 空文字列
>>> rot13("")
""
```
